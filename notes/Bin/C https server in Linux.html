C https server in Linux

.c 
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>

#define PORT 8080
#define BACKLOG 10
#define BUFFER_SIZE 1024

#include <stdio.h>

// Entry point (avoiding C runtime)
int main() {
    int server_fd, client_fd;
    struct sockaddr_in server_addr, client_addr;
    socklen_t client_len = sizeof(client_addr);
    char buffer[BUFFER_SIZE];

    // Create socket
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) goto exit;

    // Set socket options (reuse address)
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // Configure server address
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = (PORT >> 8) | (PORT << 8); // Convert to network byte order

    // Bind socket
    if (bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) goto exit;

    // Listen for connections
    if (listen(server_fd, BACKLOG) < 0) goto exit;

    while (1) {
        // Accept client connection
        client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);
        if (client_fd < 0) continue;

        // Read request
        int bytes_read = read(client_fd, buffer, BUFFER_SIZE - 1);
        if (bytes_read < 0) {
            close(client_fd);
            continue;
        }
        buffer[bytes_read] = 0; // Null-terminate
        printf("%s", buffer);

        // Simple HTTP response
        const char response[] =
            "HTTP/1.1 200 OK\r\n"
            "Content-Type: text/html\r\n"
            "Content-Length: 13\r\n"
            "\r\n"
            "Hello, World!";

        // Send response
        write(client_fd, response, sizeof(response) - 1);

        // Close client connection
        close(client_fd);
    }

exit:
    close(server_fd);
    _exit(0);
}
