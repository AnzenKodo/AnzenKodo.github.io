2) WAP to calculate area of circle and area of rectangle using abstract class.

from abc import ABC, abstractmethod
class test(ABC):
    @abstractmethod
    def area(self):
        pass
class Rec(test):
    def area(self):
        self. l=4
        self. b=2
        print(self.l*self.b)
class Cir(test):
    def area(self):
        self.r=5
        print(3.14*self.r*self.r)
R=Rec()
R. area()
C=Cir()
C. area()

Output

8
78.5

    def accept(self):
        self. name=input("Name:")

    def marks(self):

        self. m2=int(input("M2:"))

    def total(self):
        self. t=self.m1+self.m2

        print(self.name)
        print(self.t)

r. disp()
